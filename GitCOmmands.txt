git status --> for status
git init --> creates new repo
git init -b main --> new repo with main branch
git add <file name>  -->staging
git commit -m "msg" --> to commit
git commit -a -m "msg" --> to commit without staging
git log -->to get the info
git diff --> show the diff(changes) of the file before staging
git diff --staged --> show the diff(changes) of the file after staging
git rm --cached <filename>
git add . --> to stage all the files
git push -u origin main
